<div id="output"></div>


<script>
    $( function() {
        let slider = $("#slider");
        let output = document.getElementById("output");
        slider.slider({
            slide: function(event, ui) {
                output.innerHTML = slider.slider("value");
            }
        });
    } );
</script>
