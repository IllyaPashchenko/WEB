Вставка (тег Script)

Скрипт на мові JavaScript вставлений в HTML тегом <script>


<script>
    let input_text = document.getElementById("input");
    let result = document.getElementById("result_feed")
    input_text.addEventListener("keyup", function () {
        result.innerText = input_text.value.toUpperCase().split("").reverse().join("")
    });

    let ref = document.createElement("a");
    ref.href = "#";
    ref.text = "Clear";
    ref.addEventListener("click", function () {
        input_text.value = "";
        result.textContent = "";
    });

    document.body.insertBefore(ref, result)
</script>
