Система URL

Динамічно створене гіпер-посилання яке виконує скрипт мовою JavaScript при натисканні


let ref = document.createElement("a");
ref.href = "#";
ref.text = "Clear";
ref.addEventListener("click", function () {
    input_text.value = "";
    result.textContent = "";
});

document.body.insertBefore(ref, result)
